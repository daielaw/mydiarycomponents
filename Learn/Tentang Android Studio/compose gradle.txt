// Copy the common logo into Android res so it can be referenced by @drawable/logo_laksana
val copyLauncherLogoToRes by tasks.registering(Copy::class) {
    val src = file("${projectDir}/src/commonMain/composeResources/drawable/logo_laksana_50.png")
    from(src)
    into(layout.buildDirectory.dir("generated/res/logo/drawable").get().asFile)
    // Ensure the file name in res is the expected resource name used by the launcher foreground
    rename { "logo_laksana.png" }
    // Only run if source exists
    onlyIf { src.exists() }
}

tasks.named("preBuild") {
    dependsOn(copyLauncherLogoToRes)
}

// Generate version.properties file for desktop runtime version access
val versionName = libs.versions.versionName.get()
val versionCode = libs.versions.versionCode.get()
val generateDesktopVersionProperties by tasks.registering {
    val outputDir = layout.buildDirectory.dir("generated/desktop/resources")
    outputs.dir(outputDir)
    inputs.property("versionName", versionName)
    inputs.property("versionCode", versionCode)

    doLast {
        val versionFile = outputDir.get().file("version.properties").asFile
        versionFile.parentFile.mkdirs()
        versionFile.writeText("""
            |versionName=${inputs.properties["versionName"]}
            |versionCode=${inputs.properties["versionCode"]}
        """.trimMargin())
    }
}

// Make desktop source set depend on the generated properties
kotlin.sourceSets.getByName("desktopMain") {
    resources.srcDir(layout.buildDirectory.dir("generated/desktop/resources"))
}

// Ensure version properties are generated before desktop resources are compiled
tasks.matching { it.name == "desktopProcessResources" }.configureEach {
    dependsOn(generateDesktopVersionProperties)
}

// Also ensure it runs before desktop source compilation
tasks.matching { it.name.contains("compileKotlinDesktop") }.configureEach {
    dependsOn(generateDesktopVersionProperties)
}

// Generate version.js file for WASM runtime version access
val generateWasmVersionJs by tasks.registering {
    val outputDir = layout.buildDirectory.dir("generated/wasm/resources")
    outputs.dir(outputDir)
    inputs.property("versionName", versionName)
    inputs.property("versionCode", versionCode)

    doLast {
        val versionFile = outputDir.get().file("version.js").asFile
        versionFile.parentFile.mkdirs()
        versionFile.writeText("""
            |// Auto-generated version info for WASM
            |export const appVersion = {
            |    versionName: "${inputs.properties["versionName"]}",
            |    versionCode: "${inputs.properties["versionCode"]}"
            |};
            |
            |if (typeof window !== 'undefined') {
            |    window.appVersion = appVersion;
            |}
        """.trimMargin())
    }
}

// Make WASM source set depend on the generated version file
kotlin.sourceSets.getByName("wasmJsMain") {
    resources.srcDir(layout.buildDirectory.dir("generated/wasm/resources"))
}

// Ensure version file is generated before WASM resources are compiled
tasks.matching { it.name == "wasmJsProcessResources" }.configureEach {
    dependsOn(generateWasmVersionJs)
}

// Also ensure it runs before WASM source compilation
tasks.matching { it.name.contains("compileKotlinWasmJs") }.configureEach {
    dependsOn(generateWasmVersionJs)
}
